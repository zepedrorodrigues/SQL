-- Create tables

CREATE TABLE GPS_Location (
	gpsLocationID integer PRIMARY KEY,
	latitude number(7, 5) NOT NULL,
	longitude number(8, 5) NOT NULL,
	CONSTRAINT ck_longitude CHECK (longitude >= -180 AND longitude <= 180),
	CONSTRAINT ck_latitude CHECK (latitude >= -90 AND latitude <= 90)
	);

CREATE TABLE Resident (
	numCC number(8) PRIMARY KEY,
	name varchar(30) NOT NULL,
	email varchar(30) NOT NULL,
	password varchar(20) NOT NULL,
	CONSTRAINT ck_numCC CHECK (numCC > 0),
	CONSTRAINT ck_email CHECK (email LIKE '%@%.%')
	);

CREATE TABLE Permission (
	permissionID integer PRIMARY KEY,
	details varchar(30) NOT NULL
	);

CREATE TABLE Role (
	roleID integer PRIMARY KEY,
	name varchar(30) NOT NULL
	);

CREATE TABLE Role_Permission (
	roleID integer NOT NULL,
	permissionID integer NOT NULL,
	CONSTRAINT pk_role_permission PRIMARY KEY (roleID, permissionID),
	CONSTRAINT fk_permissionID FOREIGN KEY (permissionID) REFERENCES Permission (permissionID),
	CONSTRAINT fk_roleID FOREIGN KEY (roleID) REFERENCES Role (roleID)
	);

CREATE TABLE House (
	houseID integer PRIMARY KEY,
    userIDHouseOwner number(8) NOT NULL,
	floors number(2) NOT NULL,
	contractedMaxPower integer NOT NULL,
    street varchar(20) NOT NULL,
	doorNumber number(5) NOT NULL,
	apartmentNumber varchar(5),
	zipCode varchar(8) NOT NULL,
	city varchar(20) NOT NULL,
	gpsLocationID integer NOT NULL,
    CONSTRAINT fk_gpsLocationID FOREIGN KEY (gpsLocationID) REFERENCES GPS_Location (gpsLocationID),
	CONSTRAINT fk_house_owner FOREIGN KEY (userIDHouseOwner) REFERENCES Resident (numCC),
	CONSTRAINT ck_zip_code CHECK (REGEXP_LIKE(zipCode, '^[0-9]{4}-[0-9]{3}$')),
	CONSTRAINT ck_floor_min CHECK (floors > 0),
	CONSTRAINT ck_contractedMaxPower CHECK (contractedMaxPower > 0)
	);

CREATE TABLE Resident_Role (
	numCC number(8) NOT NULL,
	houseID integer NOT NULL,
	roleID integer NOT NULL,
	CONSTRAINT pk_resident_role PRIMARY KEY (numCC, houseID),
	CONSTRAINT fk_res_numCC FOREIGN KEY (numCC) REFERENCES Resident (numCC),
	CONSTRAINT fk_res_houseID FOREIGN KEY (houseID) REFERENCES House (houseID),
	CONSTRAINT fk_res_roleID FOREIGN KEY (roleID) REFERENCES Role (roleID)
	);

CREATE TABLE Room (
	roomID integer PRIMARY KEY,
	name varchar(20) NOT NULL,
	floor number(2) NOT NULL,
	height number(5, 2),
	width number(5, 2) NOT NULL,
	length number(5, 2) NOT NULL,
	houseID integer NOT NULL,
	CONSTRAINT ck_width CHECK (width > 0),
	CONSTRAINT ck_length CHECK (length > 0),
	CONSTRAINT ck_height CHECK (height > 0),
	CONSTRAINT fk_houseID FOREIGN KEY (houseID) REFERENCES House (houseID)
	);

CREATE TABLE Room_Owner (
	roomID integer NOT NULL,
	roomOwnerUserID number(8) NOT NULL,
	CONSTRAINT pk_owner_room PRIMARY KEY (roomID, roomOwnerUserID),
	CONSTRAINT fk_roomID FOREIGN KEY (roomID) REFERENCES Room (roomID),
	CONSTRAINT fk_room_ownerID FOREIGN KEY (roomOwnerUserID) REFERENCES Resident (numCC)
	);

CREATE TABLE Power_Source (
	powerSourceID integer PRIMARY KEY,
	maximumPowerOutput integer NOT NULL,
	storageCapacity integer,
	houseID integer NOT NULL,
	CONSTRAINT ck_storageCapacity CHECK (storageCapacity > 0),
	CONSTRAINT ck_maximumPowerOutput CHECK (maximumPowerOutput > 0),
	CONSTRAINT fk_power_houseID FOREIGN KEY (houseID) REFERENCES House (houseID)
	);

CREATE TABLE Energy_Production (
	powerSourceID integer NOT NULL,
	startPeriod timestamp(0) NOT NULL,
	endPeriod timestamp(0) NOT NULL,
	energyProducedkWh integer NOT NULL,
	CONSTRAINT pk_energy_production PRIMARY KEY (powerSourceID, startPeriod),
	CONSTRAINT ck_start_end_period CHECK (endPeriod > startPeriod),
	CONSTRAINT ck_energyProducedkWh CHECK (energyProducedkWh > 0),
	CONSTRAINT fk_power_sourceID FOREIGN KEY (powerSourceID) REFERENCES Power_Source (powerSourceID)
	);

CREATE TABLE Device_Type (
	deviceTypeID integer PRIMARY KEY,
	name varchar(20)
	);

CREATE TABLE Sensor_Type (
	sensorTypeID integer PRIMARY KEY,
	name varchar(20) NOT NULL,
	unit varchar(20)
	);

CREATE TABLE Actuator_Type (
    actuatorTypeID integer PRIMARY KEY,
	name varchar(20) NOT NULL
	);

CREATE TABLE Characteristics (
	characteristicID integer PRIMARY KEY,
    deviceTypeID integer NOT NULL,
	name varchar(20) NOT NULL,
	value number(20) NOT NULL,
	unit varchar(20),
	CONSTRAINT fk_device_typeID FOREIGN KEY (deviceTypeID) REFERENCES Device_Type (deviceTypeID)
	);

CREATE TABLE Device (
	deviceID integer PRIMARY KEY,
	deviceName varchar(20) NOT NULL,
	roomID integer NOT NULL,
	deviceTypeID integer NOT NULL,
	CONSTRAINT fk_dev_roomID FOREIGN KEY (roomID) REFERENCES Room (roomID),
	CONSTRAINT fk_dev_device_typeID FOREIGN KEY (deviceTypeID) REFERENCES Device_Type (deviceTypeID)
	);

CREATE TABLE Sensor (
	sensorID integer PRIMARY KEY,
	sensorTypeID integer NOT NULL,
    deviceID integer NOT NULL,
	startOperationDate timestamp(0) NOT NULL,
	stopOperationDate timestamp(0),
	CONSTRAINT ck_start_and_stop CHECK (stopOperationDate > startOperationDate),
	CONSTRAINT fk_sensor_typeID FOREIGN KEY (sensorTypeID) REFERENCES Sensor_Type (sensorTypeID),
    CONSTRAINT fk_sens_deviceid FOREIGN KEY (deviceID) REFERENCES Device (deviceID)
	);

CREATE TABLE Readings (
	sensorID integer NOT NULL,
	time timestamp(0) NOT NULL,
	reading integer NOT NULL,
	CONSTRAINT pk_readings PRIMARY KEY (sensorID, time),
	CONSTRAINT fk_sensorID FOREIGN KEY (sensorID) REFERENCES Sensor (sensorID)
	);

CREATE TABLE Actuator (
	actuatorID integer PRIMARY KEY,
    deviceID integer NOT NULL,
    actuatorTypeID integer NOT NULL,
    CONSTRAINT fk_actType FOREIGN KEY (actuatorTypeID) REFERENCES Actuator_Type(actuatorTypeID),
    CONSTRAINT fk_deviceid FOREIGN KEY (deviceID) REFERENCES Device(deviceID)
    );

CREATE TABLE Program (
	programID integer PRIMARY KEY,
	duration integer NOT NULL,
	description varchar(50) NOT NULL,
	deviceTypeID integer NOT NULL,
	CONSTRAINT ck_duration CHECK (duration > 0),
	CONSTRAINT fk_prog_device_typeID FOREIGN KEY (deviceTypeID) REFERENCES Device_Type (deviceTypeID)
	);

CREATE TABLE Active_Periods (
	activePeriodID integer PRIMARY KEY,
	scheduled number(1) NOT NULL,
	programID integer,
	deviceID integer NOT NULL,
	startTime timestamp(0) NOT NULL,
	endTime timestamp(0) NOT NULL,
	userIDTurnedOn number(8) NOT NULL,
	energyConsumedkWh integer,
	CONSTRAINT ck_act_start_end CHECK (endTime > startTime),
	CONSTRAINT fk_act_userID_ON FOREIGN KEY (userIDTurnedOn) REFERENCES Resident (numCC),
	CONSTRAINT fk_act_deviceID FOREIGN KEY (deviceID) REFERENCES Device (deviceID),
	CONSTRAINT fk_act_programID FOREIGN KEY (programID) REFERENCES Program (programID)
	);
